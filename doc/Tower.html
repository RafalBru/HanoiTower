    class Tower

    Klasa reprezentuje wieże Hanoi.

    Klasa przechowuje informacje o wieżach (A,B,C), dyskach i liczbie ruchów.

    Klasa umożliwia realizację algorytmu układania wieży Hanoi dla podanej przez użytkownika liczby dysków.

def initialize(disks)
  @towers = {
    :A => (1..disks).to_a.reverse,
    :B => [],
    :C => []
  }
  @disks = disks
  @count_move = 0
end

def display
  print "\e[?25l"
  max_disk = @disks
  system('cls') || system('clear')
  (max_disk.downto(1)).each do |level|
    @towers.each do |tower, disks|
      if disks.size >= level
        disk_size = disks[level - 1]
        disk = '=' * (2 * disk_size - 1)  
        print disk.center(9) || '|'
      else
        print ' '.center(9)
      end
      print '|'
    end
    puts
  end
  puts '    A         B         C '.center(9)
  puts "Liczba ruchow: #{@count_move}"
  sleep(0.7)
end

def hanoi(n, from, add, to)
  if n == 1
    move_disk(from, to)
    display
  else
    hanoi(n - 1, from, to, add)
    move_disk(from, to)
    display
    hanoi(n - 1, add, from, to)
  end
end

def move_disk(from, to)
  disk = @towers[from].pop
  @towers[to].push(disk)
  @count_move += 1
end
